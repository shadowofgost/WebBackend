SELECT
  t_cyplan.ID,
  t_cyplan.Rem,
  t_cyplan.Introduction,
  t_cyplan.TimeUpdate,
  t_cyplan.IdManager,
  t_cyplan.IMark,
  t_cyplan.back_up1,
  t_cyplan.back_up2,
  t_cyplan.back_up3,
  t_cyplan.back_up4,
  t_cyplan.back_up5,
  t_cyplan.back_up6,
  t_cyplan.ID_Curricula,
  t_cyplan.ID_Location,
  t_cyplan.ID_Speaker,
  t_cyplan.TimeBegin,
  t_cyplan.TimeEnd,
  t_cyplan.Attr,
  t_cyplan.Charge,
  t_cyplan.PwAccess,
  t_cyplan.PwContinuous,
  t_cyplan.PwDirection,
  t_cyplan.DoorOpen,
  t_cyplan.TimeBeginCheckBegin,
  t_cyplan.TimeBeginCheckEnd,
  t_cyplan.TimeEndCheckBegin,
  t_cyplan.TimeEndCheckEnd,
  t_cyplan.RangeUsers,
  t_cyplan.ListDepts,
  t_cyplan.RangeEqus,
  t_cyplan.ListPlaces,
  t_cyplan.MapUser2Equ,
  t_cyplan.AboutSpeaker
FROM
  t_cyplan,
  t_cyuser,
  t_cycurricula,
  t_cylocation
WHERE
  t_cyplan.ID_Speaker = t_cyuser.ID
  AND t_cyplan.ID_Curricula = t_cycurricula.ID
  AND t_cyplan.ID_Location = t_cylocation.ID
